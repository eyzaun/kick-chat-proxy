# Check current branch
git branch

# Check remote repositories 
git remote -v

# If you need to change the remote URL
git remote set-url origin https://github.com/eyzaun/kick-chat-proxy

# Push to master branch (since that's what you're on)
git push -u origin master

# Alternatively, if you want to use main branch:
git branch -M main
git push -u origin main
